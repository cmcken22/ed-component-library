import { Meta, ColorPalette, ColorItem } from "@storybook/blocks";
import { palette } from "./palette";

export const RenderPalette = (variant) => {
  return palette[variant];
};

export const RenderFullPalette = () => {
  const res = [];
  for (const key in palette) {
    if (key === "action") continue;
    if (typeof palette[key] !== "object") continue;
    res.push(<ColorItem title={key} colors={{ ...RenderPalette(key) }} />);
  }
  return res;
};

<Meta title="System/Colors" />

<ColorPalette>{RenderFullPalette()}</ColorPalette>
